```jsx
// src/components/Markdown/MarkdownRenderer.jsx
import ReactMarkdown from 'react-markdown';
import { remarkGfm } from 'remark-gfm';
import { rehypeHighlight } from 'rehype-highlight';
import CodeBlock from './CodeBlock';
import ApiTable from './ApiTable';
import TableOfContents from './TableOfContents';

const MarkdownRenderer = ({ content }) => {
  return (
    <div className="markdown-content">
      <ReactMarkdown
        remarkPlugins={[remarkGfm]}
        rehypePlugins={[rehypeHighlight]}
        components={{
          // 自定义代码块
          code: ({ node, inline, className, children, ...props }) => {
            return inline ? (
              <code className={className} {...props}>
                {children}
              </code>
            ) : (
              <CodeBlock className={className} {...props}>
                {children}
              </CodeBlock>
            );
          },
          // 自定义表格
          table: ({ children }) => (
            <ApiTable>{children}</ApiTable>
          ),
          // 自定义链接
          a: ({ href, children }) => (
            <a href={href} target="_blank" rel="noopener noreferrer">
              {children}
            </a>
          ),
          // 自定义标题（用于生成目录）
          h2: ({ children, ...props }) => (
            <h2 id={generateId(children)} {...props}>
              {children}
            </h2>
          ),
          h3: ({ children, ...props }) => (
            <h3 id={generateId(children)} {...props}>
              {children}
            </h3>
          ),
        }}
      >
        {content}
      </ReactMarkdown>
    </div>
  );
};

// 生成标题ID
const generateId = (text) => {
  return text
    .toString()
    .toLowerCase()
    .replace(/\s+/g, '-')
    .replace(/[^\w\-]+/g, '')
    .replace(/\-\-+/g, '-')
    .replace(/^-+/, '')
    .replace(/-+$/, '');
};

export default MarkdownRenderer;
```

### 4.2 代码块组件

```jsx
// src/components/Markdown/CodeBlock.jsx
import { useState } from 'react';
import hljs from 'highlight.js';

const CodeBlock = ({ children, className, ...props }) => {
  const [copied, setCopied] = useState(false);
  
  // 提取语言
  const language = className?.replace('language-', '') || 'javascript';
  
  // 语法高亮
  const highlightedCode = hljs.highlight(children.toString(), {
    language,
  }).value;
  
  // 复制功能
  const handleCopy = async () => {
    try {
      await navigator.clipboard.writeText(children.toString());
      setCopied(true);
      setTimeout(() => setCopied(false), 2000);
    } catch (err) {
      console.error('复制失败:', err);
    }
  };
  
  return (
    <div className="code-block relative">
      {/* 代码块头部 */}
      <div className="code-header flex justify-between items-center bg-gray-800 px-4 py-2 rounded-t-lg">
        <span className="language-tag text-sm text-gray-400">
          {language}
        </span>
        <button
          onClick={handleCopy}
          className="copy-btn text-sm text-gray-400 hover:text-white"
        >
          {copied ? '✓ 已复制' : '复制'}
        </button>
      </div>
      
      {/* 代码内容 */}
      <pre className="bg-gray-900 p-4 rounded-b-lg overflow-x-auto">
        <code
          className={`hljs language-${language}`}
          dangerouslySetInnerHTML={{ __html: highlightedCode }}
        />
      </pre>
    </div>
  );
};

export default CodeBlock;
```

### 4.4 目录导航组件

```jsx
// src/components/Markdown/TableOfContents.jsx
import { useState, useEffect } from 'react';

const TableOfContents = ({ content }) => {
  const [headings, setHeadings] = useState([]);
  const [activeId, setActiveId] = useState('');
  
  // 从Markdown内容提取标题
  useEffect(() => {
    const extracted = extractHeadings(content);
    setHeadings(extracted);
  }, [content]);
  
  // 监听滚动，高亮当前标题
  useEffect(() => {
    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach(entry => {
          if (entry.isIntersecting) {
            setActiveId(entry.target.id);
          }
        });
      },
      { rootMargin: '-20% 0% -80% 0%' }
    );
    
    headings.forEach(heading => {
      const element = document.getElementById(heading.id);
      if (element) observer.observe(element);
    });
    
    return () => observer.disconnect();
  }, [headings]);
  
  // 提取标题
  const extractHeadings = (content) => {
    const headingRegex = /^(#{2,3})\s+(.+)$/gm;
    const headings = [];
    let match;
    
    while ((match = headingRegex.exec(content)) !== null) {
      const level = match[1].length;
      const text = match[2];
      const id = generateId(text);
      
      headings.push({ id, text, level });
    }
    
    return headings;
  };
  
  // 生成标题ID
  const generateId = (text) => {
    return text
      .toLowerCase()
      .replace(/\s+/g, '-')
      .replace(/[^\w\-]+/g, '')
      .replace(/\-\-+/g, '-')
      .replace(/^-+/, '')
      .replace(/-+$/, '');
  };
  
  // 点击跳转
  const handleClick = (id) => {
    const element = document.getElementById(id);
    if (element) {
      element.scrollIntoView({ behavior: 'smooth' });
    }
  };
  
  return (
    <nav className="toc sticky top-20 max-h-[calc(100vh-6rem)] overflow-y-auto">
      <h4 className="text-lg font-semibold mb-4">目录</h4>
      <ul className="space-y-2">
        {headings.map(heading => (
          <li
            key={heading.id}
            className={`cursor-pointer hover:text-cyan-400 ${
              activeId === heading.id ? 'text-cyan-400 font-semibold' : ''
            }`}
            style={{ paddingLeft: `${(heading.level - 2) * 16}px` }}
            onClick={() => handleClick(heading.id)}
          >
            {heading.text}
          </li>
        ))}
      </ul>
    </nav>
  );
};

export default TableOfContents;
```

### 4.5 文档数据管理

```javascript
// src/stores/docsStore.js
import { create } from 'zustand';

// 导入文档内容
import helloWorld from '../data/docs/tutorials/hello-world.md?raw';
import camera from '../data/docs/api/camera/camera.md?raw';
import marker from '../data/docs/api/marker/marker.md?raw';
import tileLayer from '../data/docs/api/layer/tile-layer.md?raw';
import weather from '../data/docs/api/weather/weather.md?raw';
import settings from '../data/docs/api/settings/settings.md?raw';

// 文档数据
const docsData = {
  'hello-world': { content: helloWorld, title: '快速开始', category: 'tutorials' },
  'camera': { content: camera, title: '相机操作', category: 'api' },
  'marker': { content: marker, title: '场景标注', category: 'api' },
  'tile-layer': { content: tileLayer, title: '图层加载', category: 'api' },
  'weather': { content: weather, title: '环境天气', category: 'api' },
  'settings': { content: settings, title: '系统设置', category: 'api' },
};

const useDocsStore = create((set, get) => ({
  docs: docsData,
  currentDoc: null,
  loading: false,
  error: null,
  
  // 获取文档
  getDoc: (slug) => {
    const { docs } = get();
    return docs[slug] || null;
  },
  
  // 获取所有文档列表
  getDocsList: () => {
    const { docs } = get();
    return Object.entries(docs).map(([slug, doc]) => ({
      slug,
      title: doc.title,
      category: doc.category,
    }));
  },
  
  // 按分类获取文档
  getDocsByCategory: (category) => {
    const { docs } = get();
    return Object.entries(docs)
      .filter(([_, doc]) => doc.category === category)
      .map(([slug, doc]) => ({
        slug,
        title: doc.title,
      }));
  },
}));

export default useDocsStore;
```

### 4.6 更新文档页面

```jsx
// src/pages/DocPage.jsx
import { useParams } from 'react-router-dom';
import { useState, useEffect } from 'react';
import MarkdownRenderer from '../components/Markdown/MarkdownRenderer';
import TableOfContents from '../components/Markdown/TableOfContents';
import useDocsStore from '../stores/docsStore';

const DocPage = () => {
  const { slug } = useParams();
  const [content, setContent] = useState('');
  const [title, setTitle] = useState('');
  const { getDoc } = useDocsStore();
  
  useEffect(() => {
    const doc = getDoc(slug);
    if (doc) {
      setContent(doc.content);
      setTitle(doc.title);
    } else {
      setContent('# 文档不存在');
      setTitle('404');
    }
  }, [slug, getDoc]);
  
  return (
    <div className="flex gap-8">
      {/* 文档内容 */}
      <article className="flex-1 max-w-4xl">
        <h1 className="text-3xl font-bold mb-6">{title}</h1>
        <MarkdownRenderer content={content} />
      </article>
      
      {/* 目录导航 */}
      <aside className="w-64 hidden lg:block">
        <TableOfContents content={content} />
      </aside>
    </div>
  );
};

export default DocPage;
```

## 七、常见问题

### Q1: Markdown不渲染
A: 检查react-markdown版本，确保插件正确安装

A: 检查标题ID生成逻辑，确保唯一性
