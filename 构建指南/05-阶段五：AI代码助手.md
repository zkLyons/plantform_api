// src/utils/ai/api-whitelist.js
// 从文档中提取API方法，构建白名单
export const buildApiWhitelist = (docs) => {
  const whitelist = new Set();
  
  Object.entries(docs).forEach(([slug, doc]) => {
    // 提取命名空间
    const namespace = extractNamespace(slug);
    if (namespace) {
      whitelist.add(`fdapi.${namespace}`);
    }
    
    // 提取方法
    const methods = extractMethods(doc.content);
    methods.forEach(method => {
      whitelist.add(`fdapi.${namespace}.${method}`);
    });
  });
  
  return whitelist;
};

// 提取命名空间
const extractNamespace = (slug) => {
  const namespaceMap = {
    'camera': 'camera',
    'marker': 'marker',
    'tile-layer': 'tileLayer',
    'weather': 'weather',
    'settings': 'settings',
    'tools': 'tools',
    'coord': 'coord',
    'misc': 'misc',
    'polygon': 'vector',
    'polyline': 'vector',
    'heatmap': 'overlay',
  };
  
  return namespaceMap[slug] || null;
};

// 提取方法名
const extractMethods = (content) => {
  const methodRegex = /### (\w+)/g;
  const methods = [];
  let match;
  
  while ((match = methodRegex.exec(content)) !== null) {
    methods.push(match[1]);
  }
  
  return methods;
};

// 验证代码中的API调用
export const validateApiCalls = (code, whitelist) => {
  const apiCallRegex = /fdapi\.\w+\.\w+/g;
  const calls = code.match(apiCallRegex) || [];
  
  const invalidCalls = calls.filter(call => !whitelist.has(call));
  
  return {
    valid: invalidCalls.length === 0,
    invalidCalls,
    totalCalls: calls.length,
  };
};

// src/utils/ai/ai-service.js
import { AI_PROVIDERS } from './providers';

export class AIService {
  constructor(providerId, apiKey, model) {
    this.provider = AI_PROVIDERS[providerId];
    this.apiKey = apiKey;
    this.model = model || this.provider.defaultModel;
    this.abortController = null;
  }
  
  // 发送消息（流式）
  async sendMessageStream(messages, systemPrompt, onChunk, onDone, onError) {
    this.abortController = new AbortController();
    
    const request = this.provider.formatRequest(
      [
        { role: 'system', content: systemPrompt },
        ...messages,
      ],
      this.model,
      this.apiKey
    );
    
    try {
      const response = await fetch(request.url, {
        method: 'POST',
        headers: request.headers,
        body: JSON.stringify(request.body),
        signal: this.abortController.signal,
      });
      
      if (!response.ok) {
        const error = await response.json().catch(() => ({}));
        throw new Error(error.error?.message || `HTTP ${response.status}`);
      }
      
      const reader = response.body.getReader();
      const decoder = new TextDecoder();
      let fullContent = '';
      
      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        
        const chunk = decoder.decode(value, { stream: true });
        const lines = chunk.split('\n');
        
        for (const line of lines) {
          if (!line.startsWith('data: ')) continue;
          
          const data = line.slice(6).trim();
          if (data === '[DONE]') {
            onDone(fullContent);
            return;
          }
          
          try {
            const parsed = JSON.parse(data);
            const content = this.provider.parseResponse(parsed);
            
            if (content) {
              fullContent += content;
              onChunk(content, fullContent);
            }
          } catch (e) {
            // 忽略解析错误
          }
        }
      }
      
      onDone(fullContent);
    } catch (error) {
      if (error.name !== 'AbortError') {
        onError(error);
      }
    }
  }
  
  // 停止生成
  abort() {
    if (this.abortController) {
      this.abortController.abort();
      this.abortController = null;
    }
  }
}

export const SYSTEM_PROMPT = `你是 DTS Cloud 数字孪生平台 SDK 的专业代码助手。用户在调试台中工作，根据需求直接输出可运行的 JavaScript 代码。

输出规范：

2. 使用 await 处理所有异步调用，顶层 await 可直接用
3. 参数填有意义的示例值，若需要替换的 ID 用注释标注
4. 关键步骤加中文注释，代码简洁（5-20 行为宜）
5. 若问题不涉及代码（如询问概念），正常中文回答即可

// 构建带API签名的提示词
export const buildPromptWithApiSignatures = (userQuery, apiWhitelist) => {
  // 匹配相关API
  const relatedApis = findRelatedApis(userQuery, apiWhitelist);
  
  let systemPrompt = SYSTEM_PROMPT;
  
  if (relatedApis.length > 0) {
    systemPrompt += `\n\n以下是与本次问题相关的 API 真实签名，请严格按此参数顺序与类型生成，禁止臆造参数：\n`;
    relatedApis.forEach(api => {
      systemPrompt += `\n【fdapi.${api.namespace}】\n`;
      api.methods.forEach(method => {
        systemPrompt += `  fdapi.${api.namespace}.${method.name}${method.params}\n`;
      });
    });
  }
  
  return systemPrompt;
};

// 查找相关API
const findRelatedApis = (query, apiWhitelist) => {
  const keywords = {
    camera: ['相机', '视角', '镜头', '飞行', '环绕', '视点', '机位', '定位'],
    marker: ['标注', '标记', '打点', '点位', '标牌', '图标'],
    tileLayer: ['图层', '3dt', '倾斜摄影', 'bim', '模型图层'],
    weather: ['天气', '下雨', '雨', '下雪', '雪', '雾', '云', '昼夜'],
    settings: ['设置', '画质', '后处理', '后期', '特效', '抗锯齿', 'fov'],
    tools: ['测量', '测距', '测面', '绘制', '分析', '天际线', '通视'],

    overlay: ['覆盖物', '热力', 'heatmap', '热力图', '贴花', 'decal'],
  };
  
  const related = [];
  
  Object.entries(keywords).forEach(([namespace, words]) => {
    if (words.some(word => query.includes(word))) {
      related.push({
        namespace,
        methods: getMethodsForNamespace(namespace),
      });
    }
  });
  
  return related;
};

// 获取命名空间的方法

### 4.5 AI助手组件

```jsx
// src/components/AI/AIChat.jsx
import { useState, useEffect, useRef } from 'react';
import { Send, Square, Settings } from 'lucide-react';
import { AIService } from '../../utils/ai/ai-service';
import { AI_PROVIDERS } from '../../utils/ai/providers';
import { buildPromptWithApiSignatures } from '../../utils/ai/prompts';
import { validateApiCalls } from '../../utils/ai/api-whitelist';
import useDocsStore from '../../stores/docsStore';

const AIChat = ({ onInsertCode }) => {
  const [messages, setMessages] = useState([]);
  const [input, setInput] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const [provider, setProvider] = useState('claude');
  const [apiKey, setApiKey] = useState('');
  const [model, setModel] = useState('');
  const [showSettings, setShowSettings] = useState(false);
  const messagesEndRef = useRef(null);
  const { docs } = useDocsStore();
  const [aiService, setAiService] = useState(null);
  
  // 初始化AI服务
  useEffect(() => {
    const savedProvider = localStorage.getItem('ai_provider') || 'claude';
    const savedApiKey = localStorage.getItem(`ai_key_${savedProvider}`) || '';
    const savedModel = localStorage.getItem(`ai_model_${savedProvider}`) || '';
    
    setProvider(savedProvider);
    setApiKey(savedApiKey);
    setModel(savedModel);
    
    if (savedApiKey) {
      setAiService(new AIService(savedProvider, savedApiKey, savedModel));
    }
  }, []);
  
  // 自动滚动到底部
  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [messages]);
  
  // 发送消息
  const handleSend = async () => {
    if (!input.trim() || isLoading || !aiService) return;
    
    const userMessage = { role: 'user', content: input.trim() };
    setMessages(prev => [...prev, userMessage]);
    setInput('');
    setIsLoading(true);
    
    // 添加助手消息占位
    setMessages(prev => [...prev, { role: 'assistant', content: '', streaming: true }]);
    
    // 构建系统提示词
    const systemPrompt = buildPromptWithApiSignatures(input.trim(), docs);
    
    // 发送请求
    await aiService.sendMessageStream(
      messages.concat(userMessage),
      systemPrompt,
      // onChunk
      (chunk, fullContent) => {
        setMessages(prev => {
          const newMessages = [...prev];
          newMessages[newMessages.length - 1] = {
            role: 'assistant',
            content: fullContent,
            streaming: true,
          };
          return newMessages;
        });
      },
      // onDone
      (fullContent) => {
        setMessages(prev => {
          const newMessages = [...prev];
          newMessages[newMessages.length - 1] = {
            role: 'assistant',
            content: fullContent,
            streaming: false,
          };
          return newMessages;
        });
        setIsLoading(false);
        
        // 验证API调用
        const validation = validateApiCalls(fullContent, docs);
        if (!validation.valid) {
          setMessages(prev => [...prev, {
            role: 'system',
            content: `⚠️ 生成代码中包含 SDK 文档里不存在的 API：${validation.invalidCalls.join('、')}`,
          }]);
        }
      },
      // onError
      (error) => {
        setMessages(prev => [...prev, {
          role: 'error',
          content: `❌ ${error.message}`,
        }]);
        setIsLoading(false);
      }
    );
  };
  
  // 停止生成
  const handleStop = () => {
    if (aiService) {
      aiService.abort();
    }
    setIsLoading(false);
  };
  
  // 保存设置
  const handleSaveSettings = () => {
    localStorage.setItem('ai_provider', provider);
    localStorage.setItem(`ai_key_${provider}`, apiKey);
    localStorage.setItem(`ai_model_${provider}`, model);
    
    setAiService(new AIService(provider, apiKey, model));
    setShowSettings(false);
  };
  
  // 渲染消息
  const renderMessage = (message, index) => {
    const isUser = message.role === 'user';
    const isSystem = message.role === 'system';
    const isError = message.role === 'error';
    
    return (
      <div
        key={index}
        className={`mb-4 ${isUser ? 'text-right' : 'text-left'}`}
      >
        <div
          className={`inline-block max-w-[80%] p-3 rounded-lg ${
            isUser
              ? 'bg-cyan-600 text-white'
              : isSystem
              ? 'bg-yellow-600 text-white'
              : isError
              ? 'bg-red-600 text-white'
              : 'bg-gray-700 text-white'
          }`}
        >
          <pre className="whitespace-pre-wrap font-sans">{message.content}</pre>
          {message.streaming && (
            <span className="inline-block w-2 h-4 bg-white ml-1 animate-pulse" />
          )}
        </div>
        
        {/* 代码块操作按钮 */}
        {!isUser && !isSystem && !isError && message.content.includes('fdapi') && (
          <div className="mt-2 space-x-2">
            <button
              onClick={() => onInsertCode?.(message.content)}
              className="text-sm text-cyan-400 hover:text-cyan-300"
            >
              📋 插入编辑器
            </button>
          </div>
        )}
      </div>
    );
  };
  
  return (
    <div className="flex flex-col h-full bg-gray-800 rounded-lg">
      {/* 头部 */}
      <div className="flex items-center justify-between p-4 border-b border-gray-700">
        <div className="flex items-center space-x-2">
          <span className="text-xl">🤖</span>
          <span className="font-semibold">AI 代码助手</span>
          {aiService && (
            <span className="text-xs text-gray-400 px-2 py-1 bg-gray-600 rounded">
              {AI_PROVIDERS[provider]?.name}
            </span>
          )}
        </div>
        <button
          onClick={() => setShowSettings(!showSettings)}
          className="text-gray-400 hover:text-white"
        >
          <Settings size={18} />
        </button>
      </div>
      
      {/* 设置面板 */}
      {showSettings && (
        <div className="p-4 border-b border-gray-700 bg-gray-750">
          <div className="space-y-3">
            <div>
              <label className="block text-sm text-gray-400 mb-1">AI 提供商</label>
              <select
                value={provider}
                onChange={(e) => setProvider(e.target.value)}
                className="w-full bg-gray-700 text-white p-2 rounded"
              >
                {Object.entries(AI_PROVIDERS).map(([id, p]) => (
                  <option key={id} value={id}>{p.name}</option>
                ))}
              </select>
            </div>
            
            <div>
              <label className="block text-sm text-gray-400 mb-1">API Key</label>
              <input
                type="password"
                value={apiKey}
                onChange={(e) => setApiKey(e.target.value)}
                placeholder="输入 API Key"
                className="w-full bg-gray-700 text-white p-2 rounded"
              />
            </div>
            
            <div>
              <label className="block text-sm text-gray-400 mb-1">模型</label>
              <select
                value={model}
                onChange={(e) => setModel(e.target.value)}
                className="w-full bg-gray-700 text-white p-2 rounded"
              >
                {AI_PROVIDERS[provider]?.models.map(m => (
                  <option key={m.id} value={m.id}>{m.name}</option>
                ))}
              </select>
            </div>
            
            <button
              onClick={handleSaveSettings}
              className="w-full bg-cyan-600 text-white py-2 rounded hover:bg-cyan-700"
            >
              保存设置
            </button>
          </div>
        </div>
      )}
      
      {/* 消息列表 */}
      <div className="flex-1 overflow-y-auto p-4">
        {messages.length === 0 ? (
          <div className="text-center text-gray-400 mt-8">
            <p className="text-4xl mb-4">🤖</p>
            <p>你好！我是 DTS SDK AI 代码助手。</p>
            <p className="text-sm mt-2">描述你想要的场景效果，我会生成对应的 fdapi 代码。</p>
            <div className="mt-4 text-sm">
              <p>例如：</p>
              <p>• "飞行到腾讯滨海大厦并俯视"</p>
              <p>• "添加一个蓝色发光标注点"</p>
              <p>• "开启降雨+雾效"</p>
            </div>
          </div>
        ) : (
          messages.map(renderMessage)
        )}
        <div ref={messagesEndRef} />
      </div>
      
      {/* 输入框 */}
      <div className="p-4 border-t border-gray-700">
        <div className="flex space-x-2">
          <input
            type="text"
            value={input}
            onChange={(e) => setInput(e.target.value)}
            onKeyDown={(e) => e.key === 'Enter' && !e.shiftKey && handleSend()}
            placeholder={aiService ? "描述你想要的效果..." : "请先配置 API Key"}
            disabled={!aiService || isLoading}
            className="flex-1 bg-gray-700 text-white p-3 rounded-lg disabled:opacity-50"
          />
          {isLoading ? (
            <button
              onClick={handleStop}

              <Send size={20} />

};

import Playground from '../components/Sandbox/Playground';
import AIChat from '../components/AI/AIChat';

const PlaygroundPage = () => {
  const [showAI, setShowAI] = useState(false);
  const [code, setCode] = useState('');
  
  // 插入AI生成的代码
  const handleInsertCode = (generatedCode) => {
    setCode(generatedCode);
  };
  
  return (
    <div className="h-screen flex">
      {/* 调试台 */}
      <div className={`flex-1 ${showAI ? 'w-1/2' : 'w-full'}`}>
        <Playground code={code} onCodeChange={setCode} />
      </div>
      
      {/* AI助手 */}
      {showAI && (
        <div className="w-1/2 border-l border-gray-700">
          <AIChat onInsertCode={handleInsertCode} />
        </div>
      )}
      
      {/* AI助手切换按钮 */}
      <button
        onClick={() => setShowAI(!showAI)}
        className="fixed bottom-4 right-4 bg-cyan-600 text-white p-3 rounded-full shadow-lg hover:bg-cyan-700 z-50"
        title={showAI ? '隐藏AI助手' : '显示AI助手'}
      >
        🤖
      </button>
    </div>
  );
};

export default PlaygroundPage;
```

   - 代码自动插入到调试台

## 六、常见问题

### Q1: API请求失败
A: 检查API Key是否正确，检查网络连接

### Q2: 流式输出不工作
A: 检查浏览器是否支持ReadableStream，检查API是否支持流式输出

### Q3: API验证不准确
A: 检查白名单是否正确构建，检查正则表达式

### Q4: 代码插入不工作
A: 检查回调函数是否正确传递
