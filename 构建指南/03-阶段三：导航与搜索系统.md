### 4.1 完整的侧边栏组件

```jsx
// src/components/Navigation/Sidebar.jsx
import { useState } from 'react';
import { Link, useParams } from 'react-router-dom';
import { ChevronDown, ChevronRight } from 'lucide-react';
import useDocsStore from '../stores/docsStore';

const Sidebar = () => {
  const { slug } = useParams();
  const { getDocsByCategory } = useDocsStore();
  const [expandedCategories, setExpandedCategories] = useState({
    tutorials: true,
    api: true,
  });
  
  // 获取分类文档
  const tutorials = getDocsByCategory('tutorials');
  const apiDocs = getDocsByCategory('api');
  
  // 切换分类展开/折叠
  const toggleCategory = (category) => {
    setExpandedCategories(prev => ({
      ...prev,
      [category]: !prev[category],
    }));
  };
  
  // 渲染文档列表
  const renderDocsList = (docs, category) => {
    const isExpanded = expandedCategories[category];
    
    return (
      <div className="mb-4">
        <button
          onClick={() => toggleCategory(category)}
          className="flex items-center justify-between w-full p-2 text-left font-semibold text-cyan-400 hover:bg-gray-700 rounded"
        >
          <span>{category === 'tutorials' ? '开发教程' : 'API 文档'}</span>
          {isExpanded ? <ChevronDown size={16} /> : <ChevronRight size={16} />}
        </button>
        
        {isExpanded && (
          <ul className="ml-2 mt-1">
            {docs.map((doc) => (
              <li key={doc.slug} className="mb-1">
                <Link
                  to={`/docs/${doc.slug}`}
                  className={`block p-2 rounded text-sm ${
                    slug === doc.slug
                      ? 'bg-cyan-600 text-white'
                      : 'hover:bg-gray-700'
                  }`}
                >
                  {doc.title}
                </Link>
              </li>
            ))}
          </ul>
        )}
      </div>
    );
  };
  
  return (
    <aside className="w-64 bg-gray-800 p-4 overflow-y-auto h-screen sticky top-0">
      <div className="mb-6">
        <Link to="/" className="text-xl font-bold text-white">
          DTS SDK
        </Link>
      </div>
      
      {renderDocsList(tutorials, 'tutorials')}
      {renderDocsList(apiDocs, 'api')}
    </aside>
  );
};

export default Sidebar;
```

// src/components/Navigation/Breadcrumb.jsx
import { Link, useParams } from 'react-router-dom';
import { ChevronRight } from 'lucide-react';
import useDocsStore from '../stores/docsStore';

const Breadcrumb = () => {
  const { slug } = useParams();
  const { getDoc } = useDocsStore();
  
  const doc = getDoc(slug);
  
  if (!doc) return null;
  
  const breadcrumbs = [
    { label: '首页', path: '/' },
    { label: doc.category === 'tutorials' ? '开发教程' : 'API 文档', path: '/' },
    { label: doc.title, path: `/docs/${slug}` },
  ];
  
  return (
    <nav className="flex items-center space-x-2 text-sm text-gray-400 mb-6">
      {breadcrumbs.map((crumb, index) => (
        <span key={crumb.path} className="flex items-center">
          {index > 0 && <ChevronRight size={14} className="mx-2" />}
          {index === breadcrumbs.length - 1 ? (
            <span className="text-white">{crumb.label}</span>
          ) : (
            <Link to={crumb.path} className="hover:text-cyan-400">
              {crumb.label}
            </Link>
          )}
        </span>
      ))}
    </nav>
  );
};

export default Breadcrumb;

### 4.3 搜索索引构建

```javascript
// src/utils/search.js
import Fuse from 'fuse.js';

// 构建搜索索引
export const buildSearchIndex = (docs) => {
  const searchDocs = Object.entries(docs).map(([slug, doc]) => ({
    slug,
    title: doc.title,
    content: doc.content,
    category: doc.category,
    // 提取API方法名
    apiMethods: extractApiMethods(doc.content),
  }));
  
  return new Fuse(searchDocs, {
    keys: [
      { name: 'title', weight: 0.7 },
      { name: 'content', weight: 0.3 },
      { name: 'apiMethods', weight: 0.5 },
    ],
    threshold: 0.3,
    includeMatches: true,
    minMatchCharLength: 2,
  });
};

// 提取API方法名
const extractApiMethods = (content) => {
  const methodRegex = /### (\w+)/g;
  const methods = [];
  let match;
  
  while ((match = methodRegex.exec(content)) !== null) {
    methods.push(match[1]);
  }
  
  return methods.join(' ');
};

// 搜索函数
export const searchDocs = (fuse, query, filters = {}) => {
  let results = fuse.search(query);
  
  // 应用过滤器
  if (filters.category) {
    results = results.filter(r => r.item.category === filters.category);
  }
  
  // 高亮匹配内容
  return results.map(result => ({
    ...result.item,
    score: result.score,
    highlights: highlightMatches(result.matches),
  }));
};

// 高亮匹配内容
const highlightMatches = (matches) => {
  return matches.map(match => ({
    key: match.key,
    indices: match.indices,
    value: match.value,
  }));
};
```

### 4.4 搜索组件

```jsx
// src/components/Search/SearchModal.jsx
import { useState, useEffect, useRef } from 'react';
import { Search, X } from 'lucide-react';
import { useNavigate } from 'react-router-dom';
import { buildSearchIndex, searchDocs } from '../../utils/search';
import useDocsStore from '../../stores/docsStore';

const SearchModal = ({ isOpen, onClose }) => {
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);
  const [selectedIndex, setSelectedIndex] = useState(0);
  const inputRef = useRef(null);
  const navigate = useNavigate();
  const { docs } = useDocsStore();
  const [fuse, setFuse] = useState(null);
  
  // 初始化搜索索引
  useEffect(() => {
    if (docs) {
      setFuse(buildSearchIndex(docs));
    }
  }, [docs]);
  
  // 搜索
  useEffect(() => {
    if (fuse && query.trim()) {
      const searchResults = searchDocs(fuse, query);
      setResults(searchResults);
      setSelectedIndex(0);
    } else {
      setResults([]);
    }
  }, [query, fuse]);
  
  // 聚焦输入框
  useEffect(() => {
    if (isOpen && inputRef.current) {
      inputRef.current.focus();
    }
  }, [isOpen]);
  
  // 键盘事件
  const handleKeyDown = (e) => {
    switch (e.key) {
      case 'Escape':
        onClose();
        break;
      case 'ArrowUp':
        e.preventDefault();
        setSelectedIndex(prev => Math.max(0, prev - 1));
        break;
      case 'ArrowDown':
        e.preventDefault();
        setSelectedIndex(prev => Math.min(results.length - 1, prev + 1));
        break;
      case 'Enter':
        if (results[selectedIndex]) {
          navigate(`/docs/${results[selectedIndex].slug}`);
          onClose();
        }
        break;
    }
  };
  
  // 点击结果
  const handleResultClick = (slug) => {
    navigate(`/docs/${slug}`);
    onClose();
  };
  
  if (!isOpen) return null;
  
  return (
    <div className="fixed inset-0 bg-black bg-opacity-50 flex items-start justify-center pt-20 z-50">
      <div className="bg-gray-800 rounded-lg w-full max-w-2xl mx-4">
        {/* 搜索输入框 */}
        <div className="flex items-center p-4 border-b border-gray-700">
          <Search size={20} className="text-gray-400 mr-3" />
          <input
            ref={inputRef}
            type="text"
            value={query}
            onChange={(e) => setQuery(e.target.value)}
            onKeyDown={handleKeyDown}
            placeholder="搜索文档..."
            className="flex-1 bg-transparent text-white outline-none"
          />
          <button onClick={onClose} className="text-gray-400 hover:text-white">
            <X size={20} />
          </button>
        </div>
        
        {/* 搜索结果 */}
        <div className="max-h-96 overflow-y-auto">
          {results.length === 0 && query.trim() && (
            <div className="p-4 text-gray-400 text-center">
              未找到相关结果
            </div>
          )}
          
          {results.map((result, index) => (
            <div
              key={result.slug}
              className={`p-4 cursor-pointer ${
                index === selectedIndex ? 'bg-gray-700' : 'hover:bg-gray-750'
              }`}
              onClick={() => handleResultClick(result.slug)}
            >
              <div className="flex items-center justify-between mb-1">
                <h3 className="text-white font-semibold">{result.title}</h3>
                <span className="text-xs text-gray-400 px-2 py-1 bg-gray-600 rounded">
                  {result.category === 'tutorials' ? '教程' : 'API'}
                </span>
              </div>
              <p className="text-sm text-gray-400 line-clamp-2">
                {result.content.substring(0, 150)}...
              </p>
            </div>
          ))}
        </div>
        
        {/* 快捷键提示 */}
        <div className="p-4 border-t border-gray-700 text-xs text-gray-400">
          <div className="flex items-center space-x-4">
            <span>↑↓ 导航</span>
            <span>↵ 选择</span>
            <span>ESC 关闭</span>
          </div>
        </div>
      </div>
    </div>
  );
};

export default SearchModal;
```

### 4.5 搜索触发器

```jsx
// src/components/Search/SearchTrigger.jsx
import { useState, useEffect } from 'react';
import { Search } from 'lucide-react';
import SearchModal from './SearchModal';

const SearchTrigger = () => {
  const [isOpen, setIsOpen] = useState(false);
  
  // 快捷键 Ctrl+K
  useEffect(() => {
    const handleKeyDown = (e) => {
      if ((e.ctrlKey || e.metaKey) && e.key === 'k') {
        e.preventDefault();
        setIsOpen(true);
      }
    };
    
    document.addEventListener('keydown', handleKeyDown);
    return () => document.removeEventListener('keydown', handleKeyDown);
  }, []);
  
  return (
    <>
      <button
        onClick={() => setIsOpen(true)}
        className="flex items-center space-x-2 px-4 py-2 bg-gray-700 rounded-lg hover:bg-gray-600"
      >
        <Search size={16} className="text-gray-400" />
        <span className="text-gray-400">搜索文档...</span>
        <kbd className="text-xs text-gray-500 bg-gray-600 px-2 py-1 rounded">
          Ctrl+K
        </kbd>
      </button>
      
      <SearchModal isOpen={isOpen} onClose={() => setIsOpen(false)} />
    </>
  );
};

export default SearchTrigger;
```

import { useParams } from 'react-router-dom';

import MarkdownRenderer from '../components/Markdown/MarkdownRenderer';

## 六、常见问题

### Q1: 搜索不工作
A: 检查fuse.js是否正确安装，检查搜索索引是否正确构建

### Q2: 中文搜索不生效
A: 检查fuse.js配置，确保threshold设置合理

### Q3: 快捷键不生效
A: 检查事件监听器是否正确绑定，检查是否有冲突

### Q4: 侧边栏不滚动
A: 检查CSS样式，确保设置了overflow-y: auto
