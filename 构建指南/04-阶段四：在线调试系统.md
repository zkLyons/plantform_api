### 4.1 代码沙箱工具

```javascript
// src/utils/sandbox.js
export class Sandbox {
  constructor() {
    this.iframe = null;
    this.output = [];
    this.errors = [];
  }
  
  // 初始化沙箱
  init() {
    // 创建隐藏的iframe
    this.iframe = document.createElement('iframe');
    this.iframe.style.display = 'none';
    this.iframe.sandbox = 'allow-scripts';
    document.body.appendChild(this.iframe);
    
    // 初始化输出捕获
    this.setupConsoleCapture();
  }
  
  // 设置控制台捕获
  setupConsoleCapture() {
    const iframeWindow = this.iframe.contentWindow;
    
    // 重写console方法
    iframeWindow.console = {
      log: (...args) => this.captureOutput('log', args),
      error: (...args) => this.captureOutput('error', args),
      warn: (...args) => this.captureOutput('warn', args),
      info: (...args) => this.captureOutput('info', args),
      clear: () => this.clearOutput(),
    };
    
    // 捕获错误
    iframeWindow.onerror = (message, source, lineno, colno, error) => {
      this.captureError(message, error);
    };
  }
  
  // 捕获输出
  captureOutput(type, args) {
    const formatted = args.map(arg => {
      if (typeof arg === 'object') {
        return JSON.stringify(arg, null, 2);
      }
      return String(arg);
    }).join(' ');
    
    this.output.push({ type, content: formatted, timestamp: Date.now() });
  }
  
  // 捕获错误
  captureError(message, error) {
    this.errors.push({
      message,
      stack: error?.stack,
      timestamp: Date.now(),
    });
  }
  
  // 清空输出
  clearOutput() {
    this.output = [];
    this.errors = [];
  }
  
  // 执行代码
  async execute(code, context = {}) {
    this.clearOutput();
    
    try {
      // 注入上下文（如fdapi）
      const iframeWindow = this.iframe.contentWindow;
      Object.entries(context).forEach(([key, value]) => {
        iframeWindow[key] = value;
      });
      
      // 执行代码
      const result = iframeWindow.eval(code);
      
      // 处理Promise
      if (result instanceof Promise) {
        await result;
      }
      
      return {
        success: true,
        output: this.output,
        errors: this.errors,
        result,
      };
    } catch (error) {
      return {
        success: false,
        output: this.output,
        errors: [...this.errors, {
          message: error.message,
          stack: error.stack,
          timestamp: Date.now(),
        }],
        result: null,
      };
    }
  }
  
  // 销毁沙箱
  destroy() {
    if (this.iframe) {
      document.body.removeChild(this.iframe);
      this.iframe = null;
    }
  }
}
```

### 4.3 控制台组件

```jsx
// src/components/Sandbox/Console.jsx
import { useRef, useEffect } from 'react';
import { Trash2 } from 'lucide-react';

const Console = ({ output, errors, onClear }) => {
  const consoleRef = useRef(null);
  
  // 自动滚动到底部
  useEffect(() => {
    if (consoleRef.current) {
      consoleRef.current.scrollTop = consoleRef.current.scrollHeight;
    }
  }, [output, errors]);
  
  // 渲染输出项
  const renderOutputItem = (item, index) => {
    const typeColors = {
      log: 'text-gray-300',
      error: 'text-red-400',
      warn: 'text-yellow-400',
      info: 'text-blue-400',
    };
    
    return (
      <div key={index} className={`py-1 px-2 ${typeColors[item.type] || 'text-gray-300'}`}>
        <span className="text-gray-500 mr-2">{'>'}</span>
        <span className="whitespace-pre-wrap">{item.content}</span>
      </div>
    );
  };
  
  // 渲染错误项
  const renderErrorItem = (error, index) => {
    return (
      <div key={`error-${index}`} className="py-1 px-2 text-red-400">
        <span className="text-red-500 mr-2">❌</span>
        <span>{error.message}</span>
        {error.stack && (
          <pre className="text-xs text-red-300 mt-1 ml-4 overflow-x-auto">
            {error.stack}
          </pre>
        )}
      </div>
    );
  };
  
  return (
    <div className="bg-gray-900 rounded-lg overflow-hidden">
      {/* 控制台头部 */}
      <div className="flex items-center justify-between bg-gray-800 px-4 py-2">
        <span className="text-sm text-gray-400">控制台</span>
        <button
          onClick={onClear}
          className="text-gray-400 hover:text-white"
          title="清空控制台"
        >
          <Trash2 size={14} />
        </button>
      </div>
      
      {/* 控制台内容 */}
      <div
        ref={consoleRef}
        className="h-64 overflow-y-auto font-mono text-sm"
      >
        {output.length === 0 && errors.length === 0 ? (
          <div className="p-4 text-gray-500 text-center">
            运行代码后，输出将显示在这里
          </div>
        ) : (
          <>
            {output.map(renderOutputItem)}
            {errors.map(renderErrorItem)}
          </>
        )}
      </div>
    </div>
  );
};

export default Console;
```

### 4.5 调试台组件

```jsx
// src/components/Sandbox/Playground.jsx
import { useState } from 'react';
import { Play, Square, RotateCcw } from 'lucide-react';
import CodeEditor from './CodeEditor';
import Console from './Console';
import { Sandbox } from '../../utils/sandbox';
import { createFdapiMock } from '../../utils/fdapi-mock';

// 代码示例
const codeExamples = [
  {
    name: '相机定位',
    code: `// 设置相机位置
fdapi.camera.set(492035.37, 2488806.75, 402.62, -15.0, -173.0, 0.2);
console.log('相机已移动到目标位置');`,
  },
  {
    name: '添加标注',
    code: `// 添加标注点
const markerId = fdapi.marker.add({
  id: 'marker-001',
  x: 492035.37,
  y: 2488806.75,
  z: 100,
  content: '腾讯滨海大厦'
});
console.log('标注已添加:', markerId);`,
  },
  {
    name: '天气设置',
    code: `// 设置下雨效果
fdapi.weather.setRainParam({
  speed: 10,
  intensity: 0.8
});

// 设置时间
fdapi.weather.setDateTime('2024-01-15 14:30:00');
console.log('天气设置完成');`,
  },
  {
    name: '坐标转换',
    code: `// 投影坐标转经纬度
const gcs = fdapi.coord.pcs2gcs(492035.37, 2488806.75);
console.log('经纬度:', gcs);

// 经纬度转投影坐标
const pcs = fdapi.coord.gcs2pcs(116.39, 39.9);
console.log('投影坐标:', pcs);`,
  },
];

const Playground = () => {
  const [code, setCode] = useState(codeExamples[0].code);
  const [output, setOutput] = useState([]);
  const [errors, setErrors] = useState([]);
  const [isRunning, setIsRunning] = useState(false);
  const [sandbox] = useState(() => new Sandbox());
  const [fdapi] = useState(() => createFdapiMock());
  
  // 执行代码
  const handleRun = async () => {
    setIsRunning(true);
    setOutput([]);
    setErrors([]);
    
    try {
      sandbox.init();
      const result = await sandbox.execute(code, { fdapi, console });
      
      setOutput(result.output);
      setErrors(result.errors);
    } catch (error) {
      setErrors([{ message: error.message, stack: error.stack }]);
    } finally {
      setIsRunning(false);
      sandbox.destroy();
    }
  };
  
  // 停止执行
  const handleStop = () => {
    setIsRunning(false);
    sandbox.destroy();
  };
  
  // 重置代码
  const handleReset = () => {
    setCode(codeExamples[0].code);
    setOutput([]);
    setErrors([]);
  };
  
  // 清空控制台
  const handleClearConsole = () => {
    setOutput([]);
    setErrors([]);
  };
  
  // 选择示例
  const handleSelectExample = (example) => {
    setCode(example.code);
  };
  
  return (
    <div className="h-full flex flex-col">
      {/* 工具栏 */}
      <div className="flex items-center justify-between bg-gray-800 p-4 rounded-t-lg">
        <div className="flex items-center space-x-4">
          <h2 className="text-lg font-semibold">在线调试台</h2>
          
          {/* 示例选择器 */}
          <select
            onChange={(e) => handleSelectExample(codeExamples[e.target.value])}
            className="bg-gray-700 text-white px-3 py-1 rounded"
          >
            {codeExamples.map((example, index) => (
              <option key={index} value={index}>
                {example.name}
              </option>
            ))}
          </select>
        </div>
        
        <div className="flex items-center space-x-2">
          {isRunning ? (
            <button
              onClick={handleStop}
              className="flex items-center space-x-2 px-4 py-2 bg-red-600 rounded hover:bg-red-700"
            >
              <Square size={16} />
              <span>停止</span>
            </button>
          ) : (
            <button
              onClick={handleRun}
              className="flex items-center space-x-2 px-4 py-2 bg-green-600 rounded hover:bg-green-700"
            >
              <Play size={16} />
              <span>运行</span>
            </button>
          )}
          
          <button
            onClick={handleReset}
            className="flex items-center space-x-2 px-4 py-2 bg-gray-600 rounded hover:bg-gray-700"
          >
            <RotateCcw size={16} />
            <span>重置</span>
          </button>
        </div>
      </div>
      
      {/* 编辑器和控制台 */}
      <div className="flex-1 flex flex-col lg:flex-row gap-4 p-4">
        {/* 代码编辑器 */}
        <div className="flex-1">
          <CodeEditor
            value={code}
            onChange={setCode}
          />
        </div>
        
        {/* 控制台 */}
        <div className="flex-1">
          <Console
            output={output}
            errors={errors}
            onClear={handleClearConsole}
          />
        </div>
      </div>
      
      {/* 提示信息 */}
      <div className="bg-gray-800 p-4 rounded-b-lg text-sm text-gray-400">
        <p>💡 提示：点击"运行"按钮执行代码，输出将显示在右侧控制台中</p>
        <p>🔧 可用的API对象：<code className="text-cyan-400">fdapi</code>（包含camera、marker、weather等模块）</p>
      </div>
    </div>
  );
};

export default Playground;
```

## 六、常见问题

### Q1: 代码执行失败
A: 检查沙箱初始化是否正确，检查fdapi对象是否正确注入

### Q2: 控制台输出不显示
A: 检查console重写是否正确，检查输出捕获逻辑

### Q3: 编辑器不显示
A: 检查CodeMirror依赖是否正确安装

### Q4: 沙箱安全问题
A: 使用iframe sandbox属性限制权限，避免执行危险代码
